use SocialMediaDB

/************************************************\ TABLOLAR

CREATE TABLE SiteUser (
UserId INT  PRIMARY KEY,
Firstname VARCHAR(30) NOT NULL,
Lastname VARCHAR(30) NOT NULL,
Email VARCHAR(50) NOT NULL,
DateOfBirth VARCHAR(50) ,
UserPassword VARCHAR(15) NOT NULL,
UserName VARCHAR(15) NOT NULL,
Phone INT

)

CREATE TABLE UserProfile (
ProfileId INT  PRIMARY KEY,
UserId INT NOT NULL ,
FOREIGN KEY (UserId) REFERENCES SiteUser(UserId),
Relationship VARCHAR(50),
RelationShipUserId int ,
FOREIGN KEY (RelationShipUserId) REFERENCES SiteUser(UserId),
About VARCHAR(100),
JobCompany int,
FOREIGN KEY (JobCompany) REFERENCES Company(companyId),
CerrentJob VARCHAR(50),
Adress VARCHAR(50)
)



CREATE TABLE CV (
CVID INT  PRIMARY KEY,
UserId INT NOT NULL ,
FOREIGN KEY (UserId) REFERENCES SiteUser(UserId),
LastUptade VARCHAR(50),
CvDate VARCHAR(100)
CVLINK VARCHAR(MAX)

)

CREATE TABLE GROUPS (
GroupID INT  PRIMARY KEY,
CreatorId INT NOT NULL ,
FOREIGN KEY (CreatorId) REFERENCES SiteUser(UserId),
GroupName VARCHAR(50),
CreatedDate VARCHAR(50),
GroupType VARCHAR(100),
GroupStatus VARCHAR(15),
GroupDescription VARCHAR(200)

)


CREATE TABLE MemberOfGroups (
MemberId INT,
GroupId INT NOT NULL ,
FOREIGN KEY (MemberId) REFERENCES SiteUser(UserId),
FOREIGN KEY (GroupId) REFERENCES GROUPS(GroupID),
PRIMARY KEY(MemberId,GroupId),
Role VARCHAR(15),
JoinDate VARCHAR(30)

)

CREATE TABLE siteEvents (
EventId INT PRIMARY KEY,
creatorId INT NOT NULL ,
FOREIGN KEY (creatorId) REFERENCES SiteUser(UserId),
EventName VARCHAR(50),
CreatedDate VARCHAR(50),
EventType VARCHAR(100),
DateOfEvent VARCHAR(30),
EventDescription VARCHAR(200)

)



CREATE TABLE EventPeople (
UserId INT,
EventId INT NOT NULL ,
FOREIGN KEY (UserId) REFERENCES SiteUser(UserId),
FOREIGN KEY (EventId) REFERENCES siteEvents(EventId),
PRIMARY KEY(userId,EventId),
InterestStatus VARCHAR(30)

)


CREATE TABLE Company (
companyId INT PRIMARY KEY,

CompanyName VARCHAR(30),
CompanyLocation VARCHAR(30),
DateOfFoundation VARCHAR(30),
Founder VARCHAR(60)

)



CREATE TABLE CompanyType (
companyId INT ,
TypeName VARCHAR(20) ,
FOREIGN KEY(companyId) REFERENCES Company(companyId),
PRIMARY KEY(companyID,TypeName),

)


CREATE TABLE JobExperience (
ExperienceId INT PRIMARY KEY,
UserId INT NOT NULL ,
companyId INT NOT NULL ,
FOREIGN KEY (UserId) REFERENCES SiteUser(UserId),
FOREIGN KEY (companyId) REFERENCES Company(companyId),
Departman VARCHAR(30),
DateOfStart VARCHAR(30),
DateOfLastDay VARCHAR(30),

)



CREATE TABLE Jobs (
JobOfferId INT PRIMARY KEY,
companyId INT NOT NULL ,
FOREIGN KEY (companyId) REFERENCES Company(companyId),
Departman VARCHAR(30),
Salary INT,
JobLocation VARCHAR(30),
NeededPeople INT,
)



CREATE TABLE Follow (
PersonId INT ,
FollowerId INT NOT NULL ,
FOREIGN KEY (PersonId) REFERENCES SiteUser(UserId),
FOREIGN KEY (FollowerId) REFERENCES SiteUser(UserId),
DateOfFollow VARCHAR(30),
PRIMARY KEY(PersonId,FollowerId),
reg_date TIMESTAMP
)
CREATE TABLE Friend (
PersonId INT ,
Person2Id INT NOT NULL ,
FOREIGN KEY (PersonId) REFERENCES SiteUser(UserId),
FOREIGN KEY (Person2Id) REFERENCES SiteUser(UserId),
DateOfFriendship VARCHAR(30),
PRIMARY KEY(PersonId,Person2Id),
reg_date TIMESTAMP
)

CREATE TABLE Page (
PageID INT  PRIMARY KEY,
CreatorId INT NOT NULL ,
FOREIGN KEY (CreatorId) REFERENCES SiteUser(UserId),
PageName VARCHAR(50),
CreatedDate VARCHAR(50),
PageType VARCHAR(100),
PageDescription VARCHAR(200),

)

CREATE TABLE MemberOfPage (
MemberId INT,
PageId INT NOT NULL ,
FOREIGN KEY (MemberId) REFERENCES SiteUser(UserId),
FOREIGN KEY (PageId) REFERENCES Page(PageID),
PRIMARY KEY(MemberId,PageId),
Role VARCHAR(15),
JoinDate VARCHAR(30),

)
CREATE TABLE FollowedCompany (
CompanyID INT,
UserID INT,
DateOfFollow VARCHAR(50) ,
FOREIGN KEY (CompanyID) REFERENCES Company(companyId),
FOREIGN KEY (UserID) REFERENCES SiteUser(UserId),
PRIMARY KEY(CompanyID,UserID)

)

Create Table Chat(

SenderID int,
ReceiverID int,
Message VARCHAR(max),
MessageDate Varchar(20),
MessageId int Primary key,

Foreign key (SenderID) REFERENCES SiteUser(UserId),
Foreign key (ReceiverID) References SiteUser(UserId)

)

Create table Notifications (

userId int,
NotificationType Varchar(15),
NotificationDate VARCHAR(50),


Foreign key  (userId) REFERENCES SiteUser(UserId)
)

end

after insert
as begin
Declare @Count int

begin
rollback transaction
raiserror('Needed Person cannot be lower than 1 ',16,1)

end
